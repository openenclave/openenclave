maybe_build_using_clangw(oelibc)

# NOTE: This is the minimum required for musl sources.
set_property(TARGET oelibc PROPERTY C_STANDARD 99)

enclave_compile_options(oelibc PRIVATE -fno-builtin-memcpy -fno-builtin-memset
                        -ffreestanding)

if (OE_SGX)
  enclave_compile_options(oelibc PRIVATE -mrdrnd)
endif ()

if (CMAKE_C_COMPILER_ID MATCHES GNU)
  enclave_compile_options(oelibc PRIVATE -fexcess-precision=standard
                          -frounding-math)
endif ()

# Ignore warnings from musl sources. First, shared warnings:
if (CMAKE_C_COMPILER_ID MATCHES GNU
    OR CMAKE_C_COMPILER_ID MATCHES Clang
    OR USE_CLANGW)
  enclave_compile_options(
    oelibc
    PRIVATE
    -Wno-missing-braces
    -Wno-parentheses
    -Wno-unknown-pragmas
    -Wno-conversion
    -Wno-unused-parameter
    -Wno-sign-compare)
endif ()

# Second: compiler-specific warnings:
if (CMAKE_C_COMPILER_ID MATCHES GNU)
  enclave_compile_options(
    oelibc
    PRIVATE
    -Wno-jump-misses-init
    -Wno-maybe-uninitialized
    -Wno-unknown-pragmas
    -Wno-unused-but-set-variable
    -Wno-unused-function
    -Wno-unused-value
    -Wno-unused-variable)
elseif (CMAKE_C_COMPILER_ID MATCHES Clang OR USE_CLANGW)
  enclave_compile_options(
    oelibc
    PRIVATE
    -Wno-dangling-else
    -Wno-ignored-attributes
    -Wno-logical-op-parentheses
    -Wno-string-plus-int
    -Wno-typedef-redefinition
    -Wno-unneeded-internal-declaration)
endif ()

# Disable optimizations for twalk.c to avoid Clang 8 compiler bug, see #2556
if (CMAKE_C_COMPILER_ID MATCHES Clang
    AND CMAKE_C_COMPILER_VERSION VERSION_GREATER_EQUAL 8
    AND CMAKE_C_COMPILER_VERSION VERSION_LESS 9.0)
  set_source_files_properties(${MUSLSRC}/search/twalk.c PROPERTIES COMPILE_FLAGS
                                                                   -O0)
endif ()

enclave_compile_definitions(oelibc PRIVATE _XOPEN_SOURCE=700)

if (USE_DEBUG_MALLOC)
  enclave_compile_definitions(oelibc PRIVATE OE_USE_DEBUG_MALLOC)
endif ()

if (CODE_COVERAGE)
  # Enable code coverage.
  enclave_compile_options(oelibc PRIVATE -g -O0 -fprofile-arcs -ftest-coverage)
  # Link against libgcov.
  enclave_link_libraries(oelibc PRIVATE gcov)
endif ()

# This project (and its dependents) require the enclave headers and
# library, and the musl headers.
enclave_link_libraries(
  oelibc
  PUBLIC
  oesyscall
  oecore
  oelibc_includes
  PRIVATE
  oelibasm)

enclave_include_directories(
  oelibc PRIVATE ${PROJECT_BINARY_DIR}/3rdparty/musl/musl/src/include
  ${PROJECT_BINARY_DIR}/3rdparty/musl/musl/src/internal)

if (OE_SGX)
  enclave_include_directories(
    oelibc PRIVATE ${PROJECT_BINARY_DIR}/3rdparty/musl/musl/arch/x86_64)
else ()
  enclave_include_directories(
    oelibc PRIVATE ${PROJECT_BINARY_DIR}/3rdparty/musl/musl/arch/aarch64)
endif ()

set_enclave_property(TARGET oelibc PROPERTY ARCHIVE_OUTPUT_DIRECTORY
                     ${OE_LIBDIR}/openenclave/enclave)

install_enclaves(
  TARGETS
  oelibc
  oelibasm
  EXPORT
  openenclave-targets
  ARCHIVE
  DESTINATION
  ${CMAKE_INSTALL_LIBDIR}/openenclave/enclave)
